1. 테이블 생성(pokemon 테이블과 pokemon 첨부파일 테이블)
 * 실행 SQL
CREATE TABLE pokemon (
    id NUMBER PRIMARY KEY,
    name VARCHAR2(100),
    description VARCHAR2(255),
    type VARCHAR2(50),
    height NUMBER(10),
    weight NUMBER(10),
    is_favorite NUMBER(1),
    is_public NUMBER(1),
    is_notify NUMBER(1),
    variant VARCHAR2(50),
    main_image_path VARCHAR2(255)
);

CREATE SEQUENCE pokemon_seq START WITH 1 INCREMENT BY 1;


-- Pokemon Attachment 테이블
CREATE TABLE pokemon_attachment (
    id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY, -- Oracle 12c 이상
    pokemon_id NUMBER,
    pokemon_name VARCHAR2(100),
    file_name VARCHAR2(255)
);
